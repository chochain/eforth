create stable 32 allot                  \ create an array to hold 16 strings
variable si                             \ array index (default to 0)
: inc ( -- i ) si dup @ 1 rot +! ;      \ increment array idx
: s, ( adr len -- ) inc 2* >r           \ store a string (addr, len) into the table
  stable r@ 1+ array!                      \ string len
  stable r> array! ;                       \ string addr
: s@ ( i -- ) 2* >r
  stable r@ array@
  stable r> 1+ array@ ;
: st_setup
